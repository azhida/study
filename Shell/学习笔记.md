###### 以当前日期时间命名创建一个文件夹
```
dirdate=`date +%Y%m%d`_`date +%H%M%S`
mkdir $dirdate
```
###### 执行 *.sh文件 的几种方式
```
sh test.sh
./test.sh
/etc/test.sh
. test.sh
```
###### 备份文件夹及其下所有文件
```
# 命名文件夹 -- 创建日期时间文件夹
dirdate=`date +%Y%m%d`_`date +%H%M%S`
# $dirdate 为变量
bak_dir=Shell_$dirdate
mkdir $bak_dir
echo '创建了文件夹 ' $bak_dir

# 复制 Shell文件夹下所有子文件夹和文件到 新文件夹 $bak_dir 下 
cp -r Shell/* $dirdate
echo '复制 Shell文件夹下所有文件到 新文件夹 ' $bak_dir ' 下'
```
