# 代理转发

```sh
pnpm i express express-http-proxy cors
```

```js
// proxy.js
const proxy = require('express-http-proxy');
const cors = require('cors');
const app = require('express')();

const api = 'http://127.0.0.1:3000';

// app.use('/api', proxy(api));
app.use(cors());
app.use(proxy(api));

app.listen(10001, () => console.log('Proxy server started on port 10001'));
```
